# GROMACS-LS
# Compile and install a lapack to the same install prefix (and export LD_LIBRARY_PATH temporarily when compile) to avoid global lapack visibility

cmake -DGMX_MPI=off -DGMX_THREAD_MPI=off -DGMX_OPENMP=off -DGMX_SIMD=NONE -DGMX_GPU=off -DGMX_EXTERNAL_BLAS=off -DGMX_EXTERNAL_LAPACK=off -DGMX_DEFAULT_SUFFIX=off -DGMX_BINARY_SUFFIX=_ls -DCMAKE_INSTALL_PREFIX=* -DCMAKE_C_COMPILER=gcc -DCMAKE_CXX_COMPILER=g++ -DGMX_PYTHON_PACKAGE=off -DPython3_EXECUTABLE=/python3 ..

# GROMACS-FDA

cmake -DGMX_MPI=off -DGMX_THREAD_MPI=off -DGMX_OPENMP=off -DGMX_SIMD=NONE -DGMX_GPU=off -DGMX_EXTERNAL_BLAS=off -DGMX_EXTERNAL_LAPACK=off -DGMX_BUILD_FDA=ON -DGMX_DEFAULT_SUFFIX=off -DGMX_BINARY_SUFFIX=_fda -DCMAKE_INSTALL_PREFIX=* -DCMAKE_C_COMPILER=gcc -DCMAKE_CXX_COMPILER=g++ -DGMX_PYTHON_PACKAGE=off -DPython3_EXECUTABLE=/python3 ..
