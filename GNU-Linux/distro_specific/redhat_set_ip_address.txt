ip addr show

sudo vi /etc/sysconfig/network-scripts/ifcfg-*

# NAME=eno2
# DEVICE=eno2
# BOOTPROTO=dhcp
# DNS1=10.10.0.21
# DEFROUTE=yes
# IPV6INIT=yes
# IPV6_AUTOCONF=yes
# IPV6_DEFROUTE=yes
# ONBOOT=yes
# ZONE=public

sudo vi /etc/sysconfig/network-scripts/ifcfg-*

# NAME=eno2
