# debootstrap is provided in Fedora, RHEL, openSUSE, and of course, Debian and Ubuntu 

sudo mkdir /srv/debootstrap/chroot/sid/

# Sometimes --no-merged-usr is required, while --merged-usr is the default

sudo debootstrap --arch=amd64 sid /srv/debootstrap/chroot/sid/ http://deb.debian.org/debian/
sudo debootstrap --arch=amd64 --merged-usr sid /srv/debootstrap/chroot/sid/ http://deb.debian.org/debian/
sudo debootstrap --arch=amd64 --no-merged-usr sid /srv/debootstrap/chroot/sid/ http://deb.debian.org/debian/

# Set apt proxy if needed

sudo vi /srv/debootstrap/chroot/sid/etc/apt/apt.conf.d/99aptcacherng

```
Acquire::http::proxy "http://127.0.0.1:3142/";
Acquire::ftp::proxy "ftp://127.0.0.1:3142/";
```

# Upgrade

sudo chroot /srv/debootstrap/chroot/sid/

apt update && apt upgrade -y
